function generateSeatPositions(totalSeats, { seatRadius, rowHeight, graphRadius }) {
  const points = [];
  let currentRow = 0;
  const startRad = 0;
  const endRad = Math.PI;

  while (points.length < totalSeats) {
    const currentRowRadius = graphRadius - rowHeight * currentRow;
    if (currentRowRadius <= 0) break;

    const radStep = Math.atan((2.5 * seatRadius) / currentRowRadius);
    const remaining = totalSeats - points.length;
    const rowSeats = Math.min(Math.floor((endRad - startRad) / radStep), remaining - 1);
    const evenStep = rowSeats ? (endRad - startRad) / rowSeats : 0;

    for (let i = 0; i <= rowSeats; i++) {
      const angle = rowSeats ? i * evenStep + startRad : (startRad + endRad) / 2;
      points.push({ angle, row: currentRow, r: currentRowRadius });
    }
    currentRow += 1;
  }

  points.sort((a, b) => b.angle - a.angle);
  return points;
}

function SeatArc({ council, highlightKey }) {
  const width = 260;
  const height = 130;
  const cx = width / 2;
  const cy = height - 4;

  const seatRadius = 6.2;
  const rowHeight = 14.2;
  const graphRadius = height - 6;

  const positions = useMemo(
    () => generateSeatPositions(council.length, { seatRadius, rowHeight, graphRadius }),
    [council.length]
  );

  return (
    <svg width="100%" viewBox={`0 0 ${width} ${height}`} style={{ display: "block" }}>
      {positions.map((pos, i) => {
        const member = council[i];
        if (!member) return null;
        const x = cx + pos.r * Math.cos(pos.angle);
        const y = cy - pos.r * Math.sin(pos.angle);
        const f = FACTIONS[member.faction];
        const isNamed = member.named;
        const isHighlighted = isNamed && member.key === highlightKey;
        const voteColor = member.vote === "yes" ? "#2D5C3E" : member.vote === "no" ? "#8B1A1A" : f.color;
        return (
          <g key={member.id}>
            <circle
              cx={x} cy={y} r={isNamed ? seatRadius + 1 : seatRadius} fill={voteColor}
              stroke={isNamed ? "#F5F2E8" : "none"} strokeWidth={isNamed ? 1.5 : 0}
            />
            {isHighlighted && (
              <circle cx={x} cy={y} r={seatRadius + 4} fill="none" stroke="#C9A227" strokeWidth={1.5}>
                <animate attributeName="r" values={`${seatRadius + 2};${seatRadius + 5};${seatRadius + 2}`} dur="1.6s" repeatCount="indefinite" />
              </circle>
            )}
          </g>
        );
      })}
    </svg>
  );
}

export default function App() {
  const [state, setState] = useState(initialState);
  const [sceneId, setSceneId] = useState("month1_brief");
  const [view, setView] = useState("scene");

  function applyEffects(effects) {
    if (!effects) return;
    setState((prev) => {
      const next = structuredClone(prev);
      effects(next);
      return next;
    });
  }

  function goTo(id, effects) {
    applyEffects(effects);
    setSceneId(id);
    const nextScene = SCENES[id];
    setView(nextScene && nextScene.type === "chamber_action" ? "chamber" : "scene");
  }

  const majorityCount = ["dsa", "leftWfp", "progressive", "establishment", "centrist", "felder"]
    .reduce((sum, k) => sum + (FACTIONS[k] ? state.council.filter((m) => m.faction === k).length : 0), 0);
  const minorityCount = state.council.length - majorityCount;

  return (
    <div style={styles.appShell}>
      <TopBar month={state.monthLabel} monthNum={state.month} />

      <div style={styles.mapStage}>
        <DistrictMap council={state.council} />

        <div style={styles.seatArcWidget}>
          <div style={styles.seatArcWidgetHeader}>NYC COUNCIL — 51 SEATS</div>
          <SeatArc council={state.council} highlightKey={null} />
        </div>

        <div style={styles.legendOverlay}>
          <Legend majority={majorityCount} minority={minorityCount} />
        </div>

        <div style={styles.caseFileOverlay}>
          <CaseFile state={state} />
        </div>

        {view === "chamber" && (
          <div style={styles.chamberPromptOverlay}>
            {SCENES[sceneId]?.chamberPrompt || "Advance the agenda below."}
          </div>
        )}
      </div>

      {view === "scene" && (
        <div style={styles.sceneOverlay}>
          <div style={styles.sceneOverlayInner}>
            <SceneView state={state} sceneId={sceneId} goTo={goTo} />

}

            <span style={styles.factionApprName}>{FACTIONS[k].name}</span>
            <div style={styles.factionApprTrack}>
              <div style={{ ...styles.factionApprFill, width: `${v}%`, background: FACTIONS[k].color }} />
            </div>
            <span style={styles.factionApprPct}>{v}%</span>
          </div>
        ))}
      </div>
      <div style={styles.cfSection}>
        <div style={styles.cfLabel}>Favors Held</div>
        <div style={styles.chitRow}>
          {state.resources.favorReyes && <Chit label="Reyes owes you" color="#C9A227" />}
          {state.resources.favorLindqvist && <Chit label="Governor owes you" color="#8B1A1A" />}
          {!state.resources.favorReyes && !state.resources.favorLindqvist && <span style={styles.dim}>None outstanding</span>}
        </div>
      </div>
    </div>
  );
}

function Gauge({ value }) {
  const pct = Math.max(0, Math.min(100, value));
  const color = pct >= 55 ? "#2D5C3E" : pct >= 40 ? "#C9A227" : "#8B1A1A";
  return (
    <div style={styles.gaugeOuter}>
      <div style={{ ...styles.gaugeFill, width: `${pct}%`, background: color }} />
      <div style={styles.gaugeText}>{pct}%</div>
    </div>
  );
}

function RelRow({ charKey, meters }) {
  const c = CHARACTERS[charKey];
  return (
    <div style={styles.relRow}>
      <div style={{ ...styles.relName, borderLeftColor: c.color }}>{c.name.split(" ").slice(-1)[0]}</div>
      <MiniBar label="TRUST" value={meters.trust} />
      <MiniBar label="ALIGN" value={meters.align} />
    </div>
  );
}

function MiniBar({ label, value }) {
  return (
    <div style={styles.miniBarWrap}>
      <span style={styles.miniBarLabel}>{label}</span>
      <div style={styles.miniBarTrack}><div style={{ ...styles.miniBarFill, width: `${value}%` }} /></div>
    </div>
  );
}

function Chit({ label, color }) {
  return <div style={{ ...styles.chit, borderColor: color, color }}>{label}</div>;
}

function SceneView({ state, sceneId, goTo }) {
  const scene = SCENES[sceneId];
  if (!scene) return <div style={styles.sceneCard}>— End of demo build —</div>;
  if (scene.type === "negotiation") return <NegotiationScene scene={scene} state={state} goTo={goTo} />;
  return <DialogueScene scene={scene} state={state} goTo={goTo} />;
}

function DialogueScene({ scene, state, goTo }) {
  const c = CHARACTERS[scene.speaker];
  const choices = scene.choices.filter((ch) => !ch.show || ch.show(state));
  return (
    <div style={styles.sceneCard}>
      <div style={{ ...styles.speakerTag, background: c.color }}>
        <div style={styles.speakerName}>{c.name}</div>
        <div style={styles.speakerRole}>{c.role}</div>
      </div>
      <div style={styles.transcriptText}>
        {scene.lines.map((l, i) => {
          const text = typeof l === "function" ? l(state) : l;
          return <p key={i} style={styles.line}>{text}</p>;
        })}
      </div>
      <div style={styles.choiceList}>
        {choices.map((ch, i) => (
          <button key={i} style={styles.choiceBtn} onClick={() => goTo(ch.next, ch.effect)}>
            <span style={styles.choiceMark}>§</span> {ch.text}
          </button>
        ))}
      </div>
    </div>
  );
}

function NegotiationScene({ scene, state, goTo }) {
  const [selected, setSelected] = useState([]);
  const projection = useMemo(() => scene.project(state, selected), [state, selected, scene]);

  function toggle(id) {
    setSelected((s) => (s.includes(id) ? s.filter((x) => x !== id) : [...s, id]));
  }

  function submit() {
    const result = scene.resolve(state, selected);
    goTo(result.next, result.effect);
  }

  const c = CHARACTERS[scene.target];

  return (
    <div style={styles.sceneCard}>
      <div style={{ ...styles.speakerTag, background: c.color }}>
        <div style={styles.speakerName}>{c.name}</div>
        <div style={styles.speakerRole}>Negotiation — {scene.title}</div>
      </div>
      <div style={styles.transcriptText}><p style={styles.line}>{scene.ask}</p></div>
      <div style={styles.daisLabel}>ASSEMBLE YOUR OFFER</div>
      <div style={styles.offerTable}>
        {scene.offers.map((o) => {
          const valid = o.available(state);
          const active = selected.includes(o.id);
          return (
            <button
              key={o.id} disabled={!valid} onClick={() => toggle(o.id)}
              style={{ ...styles.offerChit, ...(active ? styles.offerChitActive : {}), ...(valid ? {} : styles.offerChitDisabled) }}
            >
              <div style={styles.offerChitLabel}>{o.label}</div>
              <div style={styles.offerChitCost}>{o.costLabel}</div>
            </button>
          );
        })}
      </div>
      <div style={styles.projectionBar}>
        <div style={styles.projectionLabel}>PROJECTED REACTION</div>
        <div style={styles.projectionTrack}>
          <div style={{ ...styles.projectionFill, width: `${projection.pct}%`, background: projection.pct >= 55 ? "#2D5C3E" : projection.pct >= 35 ? "#C9A227" : "#8B1A1A" }} />
        </div>
        <div style={styles.projectionNote}>{projection.note}</div>
      </div>
      <div style={styles.choiceList}>
        <button style={styles.submitBtn} onClick={submit}>PRESENT OFFER →</button>
      </div>
    </div>
  );
}
